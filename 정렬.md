# 퀵소트
```c++
void quickSort(int start, int end) {
	if (start >= end) return;

	int s = start;
	int e = end;
	int pivot = arr[(s+e)/2];

	while (s<=e)
	{
		while (arr[s] < pivot) s++;
		while (pivot < arr[e]) e--;

		if (s <= e) {
			swap(arr[s], arr[e]);
			s++; e--;
		}
	}
	quickSort(start, e);
	quickSort(s, end);
}
```

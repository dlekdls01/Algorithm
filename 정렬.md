# 삽입정렬
```c++
void insertSort(int n) {
	int num, j;
	for (int i = 1; i < n; i++) {
		num = arr[i];
		j = i - 1;

		while (j>=0 && arr[j]>num)
		{
			arr[j + 1] = arr[j];
			j--;
		}
		arr[j + 1] = num;
	}
}
```  
# 퀵정렬
```c++
void quickSort(int start, int end) {
	if (start >= end) return;

	int s = start;
	int e = end;
	int pivot = arr[(s+e)/2];

	while (s<=e)
	{
		while (arr[s] < pivot) s++;
		while (pivot < arr[e]) e--;

		if (s <= e) {
			swap(arr[s], arr[e]);
			s++; e--;
		}
	}
	quickSort(start, e);
	quickSort(s, end);
}
```
